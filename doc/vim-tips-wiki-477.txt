In vimrc:

>
    set statusline=<Whatever your status is>\t%{ShowTab()}\ %P
    fu ShowTab()
      let TabLevel = (indent('.') / &ts )
      if TabLevel == 0
        let TabLevel='*'
      endif
      return TabLevel
    endf
